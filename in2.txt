from ply.lex import lex
from ply.yacc import yacc


# Build the lexer
lexer = lex()

# Build the parser
y=yacc()

def p_error(p):
    print(f"Syntax error: ", p)
    y.success = False


# Read line from input and parse it
import sys
for linha in sys.stdin:
    y.success = True
    y.parse(linha)

    if y.success:
        print("Frase valida: ", linha)
